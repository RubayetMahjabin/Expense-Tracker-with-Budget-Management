    pip install django
    python manage.py startapp tracker
    pip install django-crispy-forms
    pip install Pillow
    pip install crispy-bootstrap4
    pip install matplotlib

#Run


    python manage.py makemigrations
    python manage.py migrate
    python manage.py runserver
